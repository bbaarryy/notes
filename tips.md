3. Не забывай про dp и то , что предпосчитать C можно след. способом:
```c++
    ll n;
    cin >> n;
    vector<vector<ll>> c(n + 1, vector<ll>(n + 1,0));
    for (int i = 0; i <= n; i++) {
        c[0][i] = 1;
        c[1][i] = i;
        c[i][i] = 1;
    }
    c[0][0] = 0;
    for (int j = 2; j <= n; j++) {
        for (int i = j+1; i <= n; i++) {
            c[j][i] = (c[j][i-1] + c[j - 1][i - 1])%MOD;
        }
    }
```
